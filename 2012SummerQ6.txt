#include <stdio.h>
#define N 6

int func(int mat[N][N])
{
	int i, j, num;															//הכרזה על שלושה משתנים מסוג מספר שלם
	int mat1[N][N] = { 0 };													//יצירת מערך באותו גודל של המערך המתקבל ואיפוסו

	mat1[0][0] = 1;															//הצבת 1 לתוך התא הראשון
	num = mat[0][0];														//הצבה של התא הראשון במערך המתקבל לתוך המשתנה
	for (i = 1; i < N; i++)
	{
		for (j = 0; j <= i; j++)											//לולאה שרצה עד למספר הסיבובים עד עכשיו כולל, הלולאה בודקת עם התא במערך המתקבל שווה לתא הראשון והתא מעלו או בצדו היה שווה עם עומד בתנאי מציב 1 אם לא משאיר מאופס
		{
			if ((j < i) && (mat1[i - 1][j] && (mat[i][j] == num)))			//בודק תאים לפי עמודות עד ששני הלולאות תואמות
				mat1[i][j] = 1;
			if ((j > 0) && (mat1[i][j - 1]) && (mat[i][j] == num))			//בודק תאים לפי עמודות גם אחרי שהלולאות תואמות
				mat1[i][j] = 1;
			if ((j < i) && (mat1[j][i - 1]) && (mat[j][i] == num))			//בודק תאים לפי שורות עד ששני הלולאות תואמות
				mat1[j][i] = 1;
			if ((j > 0) && (mat1[j - 1][i]) && (mat[j][i] == num))			//בודק תאים לפי שורות גם אחרי שהלולאות תואמות
				mat1[j][i] = 1;
		}
		if (mat[i][i] == num)												//בודק רק את התא האמצעי בין השורות לעמודות
			if (mat1[i - 1][i] || mat1[i][i - 1]) mat1[i][i] = 1;			//מאשר את התנאי
	}
	return mat1[N - 1][N - 1];												//מחזיר את התא האחרון של המערך שיצרנו אשר הוא 1 אם התנאי עמד בשורות או בעמודות ממהתחלה לסוף
}

int main()
{
	int res;
	int mat[N][N] = {{3,3,2,2,5,6},
					 {3,3,3,2,7,8},
					 {3,3,3,3,6,7},
					 {4,3,3,3,3,3},
					 {4,4,4,3,3,3},
					 {4,7,5,4,3,3}}; 			//הצבה של מערך דו מימדי
	res = func(mat); 							//קריאה לפונקציה והצבה של הערך החוזר למשתנה
	if (res) printf ("The matrix is good\n");	//הדפסה עם המערך עמד בתנאי (הבט למטה)
	else printf ("The matrix is not good\n");	//הדפסה אם התנאי לא עומד למערך
	return 0;
}
